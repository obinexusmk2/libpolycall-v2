# LibPolyCall v2 — Telemetry Code Replay System: Symbolic Interpreter Specification

**Repository:** `github.com/obinexusmk2/libpolycall-v2`  
**Component:** `polycore telemetry` subsystem  
**Derived from:** OBINexus video corpus + constitutional computing framework  
**Author directive:** CodexAI implementation target  

---

## 1. System Overview

The **Telemetry Code Replay Symbolic Interpreter (TCRSI)** is the core intelligence layer of the `polycore telemetry` command. It provides:

1. **Silent observation** — a stateless protocol observer that intercepts request/response cycles without mutating system state
2. **GUID-seeded state tracking** — cryptographically seeded global unique identifiers that encode every state transition
3. **Replay engine** — deterministic re-execution of any captured state sequence to reproduce bugs or verify correctness
4. **Symbolic interpretation** — filter/flash function mapping that transforms raw state sequences into human-readable symbolic traces

The system is grounded in the **NFA (Nondeterministic Finite Automaton)** model mapped to the polycore router binding architecture.

---

### 3.1 GUID State Node

```c
/* include/polycall/telemetry/guid.h */

#ifndef POLYCALL_TELEMETRY_GUID_H
#define POLYCALL_TELEMETRY_GUID_H

#include <stdint.h>
#include <time.h>

#define POLYCALL_GUID_SEED_LEN   32   /* SHA-256 output bytes          */
#define POLYCALL_GUID_STRING_LEN 65   /* hex string + null terminator  */
#define POLYCALL_MAX_STATE_DEPTH 4096 /* max automaton states per trace */

typedef struct polycall_guid {
    uint8_t  seed[POLYCALL_GUID_SEED_LEN];  /* cryptographic seed          */
    char     string[POLYCALL_GUID_STRING_LEN]; /* hex string representation */
    uint64_t timestamp_ns;                   /* nanosecond timestamp        */
    uint32_t sequence_id;                    /* monotonic sequence counter  */
} polycall_guid_t;

typedef enum polycall_state_type {
    STATE_REQUEST_RECEIVED  = 0x01,
    STATE_AUTH_CHECK        = 0x02,
    STATE_DATA_PROCESSED    = 0x03,
    STATE_RESPONSE_SENT     = 0x04,
    STATE_ERROR             = 0x05,
    STATE_ROUTE_TRANSITION  = 0x06,
    STATE_BINDING_INVOKED   = 0x07,
    STATE_EDGE_CACHE_HIT    = 0x08,
    STATE_EDGE_CACHE_MISS   = 0x09,
} polycall_state_type_t;

typedef struct polycall_state_node {
    polycall_guid_t          guid;          /* unique ID for this state        */
    polycall_state_type_t    type;          /* what kind of state transition   */
    char                     route[256];    /* endpoint/route that was active  */
    char                     binding[128];  /* which language binding invoked  */
    uint8_t                  action_hash[32]; /* SHA-256 of actions taken      */
    struct polycall_state_node *prev;       /* linked list: previous state     */
    struct polycall_state_node *next;       /* linked list: next state         */
    uint64_t                 timestamp_ns;  /* when this state was entered     */
} polycall_state_node_t;

typedef struct polycall_state_trace {
    polycall_state_node_t   *head;          /* oldest state in trace           */
    polycall_state_node_t   *tail;          /* most recent state               */
    uint32_t                 depth;         /* number of states recorded       */
    polycall_guid_t          trace_guid;    /* GUID for the entire trace       */
    char                     uid[64];       /* user identifier for session     */
} polycall_state_trace_t;

#endif /* POLYCALL_TELEMETRY_GUID_H */
```

### 3.2 Symbolic Filter/Flash Map

```c
/* include/polycall/telemetry/symbolic.h */

#ifndef POLYCALL_TELEMETRY_SYMBOLIC_H
#define POLYCALL_TELEMETRY_SYMBOLIC_H

#include "guid.h"

/*
 * FILTER: pure function — processes state without destroying it.
 *         Mutates data representation, not program state.
 *         (splice/search/sort/merge operations)
 *
 * FLASH:  stores the filtered result bidirectionally.
 *         Two-to-one: two inputs produce one symbol output.
 *         One-to-two: one symbol can be decoded back to both inputs.
 */

#define POLYCALL_SYMBOL_MAX_LEN   128
#define POLYCALL_FILTER_DEPTH     16   /* max filter pipeline stages */

typedef struct polycall_symbol {
    char     name[POLYCALL_SYMBOL_MAX_LEN];
    uint32_t code;
    uint8_t  hash[32];                 /* SHA-256 of symbol content  */
    int      is_bidirectional;         /* can be replayed in reverse  */
} polycall_symbol_t;

typedef struct polycall_filter_result {
    polycall_symbol_t  symbol;
    polycall_state_node_t *source_state;
    char               sparse_repr[512]; /* sparse representation of state */
    int                is_terminal;      /* true if this is an error/end   */
} polycall_filter_result_t;

/* Filter function signature: pure, does not modify source state */
typedef int (*polycall_filter_fn)(
    const polycall_state_node_t *in,
    polycall_filter_result_t    *out
);

/* Flash function: store filtered result bidirectionally */
typedef int (*polycall_flash_fn)(
    const polycall_filter_result_t *in,
    polycall_symbol_t              *out_forward,
    polycall_symbol_t              *out_inverse   /* NULL if not needed */
);

typedef struct polycall_symbolic_pipeline {
    polycall_filter_fn  filters[POLYCALL_FILTER_DEPTH];
    polycall_flash_fn   flash;
    uint32_t            filter_count;
    int                 bidirectional;   /* enable inverse relay */
} polycall_symbolic_pipeline_t;

#endif /* POLYCALL_TELEMETRY_SYMBOLIC_H */
```

## 4. GUID Engine Implementation

```c
/* src/telemetry/guid_engine.c */

#include "polycall/telemetry/guid.h"
#include <openssl/sha.h>
#include <openssl/rand.h>
#include <stdio.h>
#include <string.h>
#include <time.h>

/*
 * Seed the GUID using SHA-256 over:
 *   random_bytes || timestamp_ns || sequence_id || route
 *
 * This ensures:
 *  - No two GUIDs are the same (randomness + time)
 *  - GUIDs are reproducible given the seed (deterministic replay)
 *  - Route is bound into the GUID (state-aware)
 */
int polycall_guid_generate(
    polycall_guid_t *guid,
    const char      *route,
    uint32_t         sequence_id
) {
    if (!guid || !route) return -1;

    /* --- Step 1: gather entropy --- */
    uint8_t random_bytes[16];
    if (RAND_bytes(random_bytes, sizeof(random_bytes)) != 1) return -2;

    struct timespec ts;
    clock_gettime(CLOCK_REALTIME, &ts);
    uint64_t timestamp_ns = (uint64_t)ts.tv_sec * 1000000000ULL + ts.tv_nsec;

    /* --- Step 2: build seed material --- */
    uint8_t seed_material[256];
    size_t  offset = 0;

    memcpy(seed_material + offset, random_bytes, 16);  offset += 16;
    memcpy(seed_material + offset, &timestamp_ns, 8);  offset += 8;
    memcpy(seed_material + offset, &sequence_id, 4);   offset += 4;

    size_t route_len = strlen(route);
    if (route_len > 228) route_len = 228;
    memcpy(seed_material + offset, route, route_len);  offset += route_len;

    /* --- Step 3: SHA-256 hash --- */
    SHA256(seed_material, offset, guid->seed);

    /* --- Step 4: convert to hex string --- */
    for (int i = 0; i < POLYCALL_GUID_SEED_LEN; i++) {
        snprintf(guid->string + (i * 2), 3, "%02x", guid->seed[i]);
    }
    guid->string[POLYCALL_GUID_STRING_LEN - 1] = '\0';

    guid->timestamp_ns = timestamp_ns;
    guid->sequence_id  = sequence_id;

    return 0;
}

/*
 * Create a new state node and attach it to the trace.
 * This is called at every state transition in the polycore system.
 */
int polycall_trace_push_state(
    polycall_state_trace_t *trace,
    polycall_state_type_t   type,
    const char             *route,
    const char             *binding,
    const uint8_t          *action_data,
    size_t                  action_data_len
) {
    if (!trace) return -1;

    polycall_state_node_t *node = calloc(1, sizeof(polycall_state_node_t));
    if (!node) return -2;

    /* Generate GUID for this state */
    polycall_guid_generate(&node->guid, route, trace->depth);

    node->type = type;
    strncpy(node->route,   route   ? route   : "unknown", 255);
    strncpy(node->binding, binding ? binding : "none",    127);

    /* Hash the action data to bind it into the state */
    if (action_data && action_data_len > 0) {
        SHA256(action_data, action_data_len, node->action_hash);
    }

    struct timespec ts;
    clock_gettime(CLOCK_REALTIME, &ts);
    node->timestamp_ns = (uint64_t)ts.tv_sec * 1000000000ULL + ts.tv_nsec;

    /* Append to linked list */
    node->prev = trace->tail;
    node->next = NULL;

    if (trace->tail) {
        trace->tail->next = node;
    } else {
        trace->head = node;
    }
    trace->tail = node;
    trace->depth++;

    return 0;
}
```

## 5. Replay Engine Implementation

```c
/* src/telemetry/replay_engine.c */

#include "polycall/telemetry/guid.h"
#include "polycall/telemetry/symbolic.h"
#include <stdio.h>
#include <string.h>

typedef enum polycall_replay_direction {
    REPLAY_FORWARD  = 0,  /* trace states head → tail (normal)        */
    REPLAY_INVERSE  = 1,  /* trace states tail → head (inverse relay) */
} polycall_replay_direction_t;

typedef struct polycall_replay_result {
    polycall_state_node_t  *failure_state;  /* node where replay diverged  */
    uint32_t                states_replayed;
    char                    failure_guid[POLYCALL_GUID_STRING_LEN];
    char                    summary[1024];  /* human-readable trace summary */
    int                     reproduced;     /* 1 if failure was reproduced  */
} polycall_replay_result_t;

/*
 * Replay the captured state trace against the live system.
 *
 * For each state in the trace:
 *   1. Re-invoke the binding/route with the stored action hash
 *   2. Compare the resulting GUID against the stored GUID
 *   3. If mismatch → failure_state is set, replay stops
 *
 * Direction:
 *   REPLAY_FORWARD  → reproduce the bug path (head to tail)
 *   REPLAY_INVERSE  → verify recovery path (tail to head, inverse relay)
 */
int polycall_replay_trace(
    const polycall_state_trace_t *trace,
    polycall_replay_direction_t   direction,
    polycall_replay_result_t     *result
) {
    if (!trace || !result) return -1;

    memset(result, 0, sizeof(*result));

    polycall_state_node_t *cursor;
    if (direction == REPLAY_FORWARD) {
        cursor = trace->head;
    } else {
        cursor = trace->tail;
    }

    uint32_t step = 0;
    char snapshot_buf[2048];

    while (cursor) {
        /* Build snapshot of this state for the summary */
        snprintf(snapshot_buf, sizeof(snapshot_buf),
            "[%u] type=%d route=%s binding=%s guid=%s ts=%llu\n",
            step,
            cursor->type,
            cursor->route,
            cursor->binding,
            cursor->guid.string,
            (unsigned long long)cursor->timestamp_ns
        );
        strncat(result->summary,
                snapshot_buf,
                sizeof(result->summary) - strlen(result->summary) - 1);

        /*
         * SYMBOLIC CHECK:
         * Verify the route is still valid in the current system state.
         * If the route no longer exists or the binding hash changed,
         * this is the divergence point.
         *
         * TODO: hook into polycore routing table for live verification
         */
        if (cursor->type == STATE_ERROR) {
            result->failure_state = cursor;
            strncpy(result->failure_guid,
                    cursor->guid.string,
                    POLYCALL_GUID_STRING_LEN - 1);
            result->reproduced = 1;
            break;
        }

        result->states_replayed = ++step;

        if (direction == REPLAY_FORWARD) {
            cursor = cursor->next;
        } else {
            cursor = cursor->prev;  /* inverse relay: walk backwards */
        }
    }

    return result->reproduced ? 1 : 0;
}

/*
 * Attach a replay result to the ticket system.
 * Generates a ticket ID from the GUID + UID and writes to endpoint.
 */
int polycall_replay_create_ticket(
    const polycall_replay_result_t *result,
    const char                     *uid,
    const char                     *ticket_endpoint,
    char                           *ticket_id_out,
    size_t                          ticket_id_len
) {
    if (!result || !uid || !ticket_endpoint) return -1;

    /* Ticket ID = first 16 chars of failure GUID + "-" + UID prefix */
    snprintf(ticket_id_out, ticket_id_len,
             "PCTT-%.16s-%.8s",
             result->failure_guid,
             uid);

    fprintf(stdout,
            "[POLYCALL TICKET] id=%s endpoint=%s states=%u reproduced=%d\n",
            ticket_id_out,
            ticket_endpoint,
            result->states_replayed,
            result->reproduced);

    /*
     * TODO: POST ticket to ticket_endpoint via polycore HTTP binding
     * Body: { "ticket_id": "...", "guid": "...", "trace_summary": "..." }
     */

    return 0;
}
```

## 6. Symbolic Interpreter — Filter/Flash Pipeline

```c
/* src/telemetry/symbolic_interpreter.c */

#include "polycall/telemetry/symbolic.h"
#include <string.h>
#include <stdio.h>

/*
 * FILTER STAGE 1: Route normalization
 * Pure function — extracts route structure without modifying state.
 * "Splice" semantics: trim noise, preserve signal.
 */
static int filter_route_normalize(
    const polycall_state_node_t *in,
    polycall_filter_result_t    *out
) {
    if (!in || !out) return -1;

    out->source_state = (polycall_state_node_t *)in;

    /* Sparse representation: remove query params, keep base route */
    const char *q = strchr(in->route, '?');
    size_t base_len = q ? (size_t)(q - in->route) : strlen(in->route);
    if (base_len >= sizeof(out->sparse_repr)) base_len = sizeof(out->sparse_repr) - 1;

    strncpy(out->sparse_repr, in->route, base_len);
    out->sparse_repr[base_len] = '\0';

    return 0;
}

/*
 * FILTER STAGE 2: State type classification
 * Maps state type enum to symbolic name.
 */
static int filter_state_classify(
    const polycall_state_node_t *in,
    polycall_filter_result_t    *out
) {
    if (!in || !out) return -1;

    static const struct { polycall_state_type_t type; const char *symbol; } map[] = {
        { STATE_REQUEST_RECEIVED,  "REQ"   },
        { STATE_AUTH_CHECK,        "AUTH"  },
        { STATE_DATA_PROCESSED,    "DATA"  },
        { STATE_RESPONSE_SENT,     "RESP"  },
        { STATE_ERROR,             "ERR"   },
        { STATE_ROUTE_TRANSITION,  "ROUTE" },
        { STATE_BINDING_INVOKED,   "BIND"  },
        { STATE_EDGE_CACHE_HIT,    "CAHIT" },
        { STATE_EDGE_CACHE_MISS,   "CAMIS" },
        { 0, NULL }
    };

    for (int i = 0; map[i].symbol; i++) {
        if (map[i].type == in->type) {
            strncpy(out->symbol.name, map[i].symbol, POLYCALL_SYMBOL_MAX_LEN - 1);
            out->symbol.code = (uint32_t)in->type;
            break;
        }
    }

    out->is_terminal = (in->type == STATE_ERROR || in->next == NULL);
    return 0;
}

/*
 * FLASH: Store filtered result bidirectionally.
 * forward  = normal symbol for replay
 * inverse  = symbol for inverse relay recovery
 *
 * Two-to-one: (route_symbol, state_symbol) → flash output
 * One-to-two: flash output → can recover both on inverse relay
 */
static int flash_bidirectional(
    const polycall_filter_result_t *in,
    polycall_symbol_t              *out_forward,
    polycall_symbol_t              *out_inverse
) {
    if (!in || !out_forward) return -1;

    /* Forward flash: route + state type → composite symbol */
    snprintf(out_forward->name, POLYCALL_SYMBOL_MAX_LEN,
             "%s::%s", in->sparse_repr, in->symbol.name);
    out_forward->code = in->symbol.code;
    out_forward->is_bidirectional = 1;

    /* Inverse flash: reverse the composite for inverse relay */
    if (out_inverse) {
        snprintf(out_inverse->name, POLYCALL_SYMBOL_MAX_LEN,
                 "INV[%s::%s]", in->symbol.name, in->sparse_repr);
        out_inverse->code = ~in->symbol.code;  /* bitwise inverse code */
        out_inverse->is_bidirectional = 1;
    }

    return 0;
}

/*
 * Run the full symbolic pipeline over a state trace.
 * Produces a symbol sequence that can be:
 *   - printed as a human-readable trace
 *   - diffed against a known-good trace
 *   - replayed in forward or inverse direction
 */
int polycall_symbolic_interpret_trace(
    const polycall_state_trace_t    *trace,
    polycall_symbolic_pipeline_t    *pipeline,
    polycall_symbol_t               *symbols_out,
    uint32_t                        *symbols_count_out,
    uint32_t                         max_symbols
) {
    if (!trace || !pipeline || !symbols_out || !symbols_count_out) return -1;

    polycall_state_node_t *cursor = trace->head;
    uint32_t count = 0;

    while (cursor && count < max_symbols) {
        polycall_filter_result_t fres;
        memset(&fres, 0, sizeof(fres));

        /* Run all filter stages */
        for (uint32_t f = 0; f < pipeline->filter_count; f++) {
            if (pipeline->filters[f](cursor, &fres) != 0) {
                return -2;
            }
        }

        /* Flash to output */
        polycall_symbol_t inv;
        pipeline->flash(&fres, &symbols_out[count], pipeline->bidirectional ? &inv : NULL);

        count++;
        cursor = cursor->next;
    }

    *symbols_count_out = count;
    return 0;
}

/*
 * Build the default pipeline for libpolycall-v2
 */
int polycall_telemetry_build_default_pipeline(
    polycall_symbolic_pipeline_t *pipeline
) {
    if (!pipeline) return -1;

    memset(pipeline, 0, sizeof(*pipeline));

    pipeline->filters[0]    = filter_route_normalize;
    pipeline->filters[1]    = filter_state_classify;
    pipeline->filter_count  = 2;
    pipeline->flash         = flash_bidirectional;
    pipeline->bidirectional = 1;

    return 0;
}
```

```c
/* src/cli/commands/telemetry_cmd.c */

#include "polycall/telemetry/guid.h"
#include "polycall/telemetry/symbolic.h"
#include <stdio.h>
#include <string.h>

/*
 * polycall telemetry --observe <endpoint>
 *   Start silent observation on an endpoint.
 *
 * polycall telemetry --replay <trace_guid>
 *   Replay a captured trace by its GUID string.
 *
 * polycall telemetry --replay <trace_guid> --inverse
 *   Replay in inverse relay mode (backwards, recovery verification).
 *
 * polycall telemetry --ticket <trace_guid> --uid <user_id> --endpoint <url>
 *   Create a ticket from a captured error trace.
 *
 * polycall telemetry --dump <trace_guid>
 *   Print symbolic interpretation of a trace.
 */

typedef enum telemetry_cmd_mode {
    TELEM_MODE_OBSERVE,
    TELEM_MODE_REPLAY,
    TELEM_MODE_TICKET,
    TELEM_MODE_DUMP,
} telemetry_cmd_mode_t;

    telemetry_cmd_mode_t mode = TELEM_MODE_DUMP;
```

## 10. Design Invariants (Constitutional Constraints)

These rules are non-negotiable per the OBINexus zero-trust constitutional computing framework:

1. **Observer is silent** — the telemetry layer NEVER modifies request/response data. It is read-only at the protocol layer.
2. **GUID seeding is mandatory** — any GUID not seeded with cryptographic entropy (minimum SHA-256) is invalid and must be rejected.
3. **MD5 is prohibited** — use SHA-256 minimum, SHA-3 preferred for new implementations.
4. **Filter functions are pure** — they mutate data representations only, never program state.
5. **Flash is bidirectional** — every flash operation must support inverse relay to enable recovery-path verification.
6. **Replay is deterministic** — the same trace GUID must always produce the same symbol sequence given the same system state.
7. **Tickets are GUID-bound** — no ticket may be created without an attached trace GUID and user UID.
8. **Production mode exclusion** — the interactive/debug replay mode is available only in `POLYCALL_BUILD_DEVELOPMENT=ON`. It MUST NOT be compiled into production builds.
